// Load certificates.json and render all certificates dynamically fetch("assets/certificates/certificates.json") .then(response => response.json()) .then(files => { const grid = document.getElementById("researchGrid"); files.forEach(file => { const card = document.createElement("div"); card.classList.add("research-img-card"); // Format name for caption const name = file .replace(/\.(jpg|jpeg|png)$/i, "") .replace(/[_\-]+/g, " ") .trim(); card.innerHTML = `

${name}

`; grid.appendChild(card); }); }) .catch(err => { console.error("Error loading certificates:", err); });
